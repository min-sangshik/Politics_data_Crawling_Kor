	function settingYN(e) {
		var electionType = $('#electionType').val();
		var electionName = $('#electionName').val();
		var electionCode = $('#electionCode').val();
		var cityCode = $('#cityCode').val();
		var townCode = $('#townCode').val();
		var sggCityCode = $('#sggCityCode').val();
		var sggTownCode = $('#sggTownCode').val();
		var proportionalRepresentationCode = $('#proportionalRepresentationCode').val();

		if(electionCode == 3 || electionCode == 11){
			appendAllOption('cityCode');
			//removeOption('cityCode',-1);
			//cityCode = $('#cityCode').val();
		}else{
			removeAllOption('cityCode');
		}

		if (electionCode == 1){
			if (electionName == -1){
				selectBoxShow("none", "none", "none", "none", "none");
			}else{
				selectBoxShow("none", "none", "none", "none", "");
			}
		}else if (electionCode == 2){
			$('#spanProportionalRepresentationCode').css('display', "none");
			if (electionName == -1){
				selectBoxShow("none", "none", "none", "none", "none");
			}else {
				$('#spanElectionCode').css('display', "");
				if (cityCode == -1){
					selectBoxShow("", "none", "none", "none", "none");
				}else {
					selectBoxShow("", "none", "none", "none", "");
				}
			}
		}else if (electionCode == 7){
			$('#spanCityCode').css('display', "none");
			if(electionName==-1){
				selectBoxShow("none", "none", "none", "none", "none");
				$('#spanProportionalRepresentationCode').css('display', "none");
			}
			else{
				$('#spanElectionCode').css('display', "");
				$('#spanProportionalRepresentationCode').css('display', "");
				if(proportionalRepresentationCode == -1){
					selectBoxShow("none", "none", "none", "none", "none");
				}else {
					selectBoxShow("none", "none", "none", "none", "");
				}
			}
		}else if (electionCode == 3 || electionCode == 8 || electionCode == 11){
			$('#spanElectionCode').css('display', "");
			$('#spanProportionalRepresentationCode').css('display', "none");
			if(cityCode >= 0){
				selectBoxShow("", "none", "none", "none", "");
			}else {
				selectBoxShow("", "none", "none", "none", "none");
			}
		}else if (electionCode == 4 || electionCode == 9 || electionCode == 10){
			$('#spanElectionCode').css('display', "");
			$('#spanProportionalRepresentationCode').css('display', "none");
			if(cityCode >= 0){
				if(sggCityCode >= 0){
					selectBoxShow("", "none", "", "none", "");
				}else {
					selectBoxShow("", "none", "", "none", "none");
				}
			}else {
				selectBoxShow("", "none", "none", "none", "none");
			}
		}else if (electionCode == 5 || electionCode == 6){
			$('#spanElectionCode').css('display', "");
			$('#spanProportionalRepresentationCode').css('display', "none");
			//alert("cityCode=" + cityCode + "   townCode=" + townCode);
			if(cityCode >= 0){
				if(townCode >= 0){
					if(sggTownCode >= 0 ){
						selectBoxShow("", "", "none", "", "");
					}else {
						selectBoxShow("", "", "none", "", "none");
					}
				}else {
					selectBoxShow("", "", "none", "none", "none");
				}
			}else {
				selectBoxShow("", "none", "none", "none", "none");
			}
		}else {
			selectBoxShow("none", "none", "none", "none", "none");
		}

		if(electionCode >= 0){
			$('#wub_select').css('display','');
		}
		
		if(electionCode == '' || electionCode == null){
			$('#electionCode').val(-1);
		}
	}
